Here the validations are done by the function validate() using JavaScript. User is alerted if any of the field in the form is left empty. Try it out yourself. Check out the code below: 

            function validate() {
            var name = document.form.name.value.length;
            var sex = document.form.sex.value.length;
            var email = document.form.email.value.length;
            var date = document.form.date.value.length;
            if (name == 0) {
                alert("Enter your Name");
                return false;
            }
            if (sex == 0) {
                alert("Select your Gender");
                return false;
            }
            if (email == 0) {
                alert("email cannot be empty");
                return false;
            }
            if (date == 0) {
                alert("Enter date");
                return false;
            }
            return true;
