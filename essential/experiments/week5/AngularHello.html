How it is done?

        <input ng-model="hello" />
        {{hello}}
Above code binds the input box to {hello} which can change dynamically. So as the content of text box changes, content also changes

Below is the code how I created the AngularHelloApp controller.

        
        var app = angular.module("AngularHelloApp", []);
        app.controller("AngularHelloController",
        function ($scope) {
        
        var employee = {
        first: "Alice",
        last: "Wonderland"
        };
        $scope.employee = employee;
        var employees = [
        { first: "Bhumitra", last: "Nagar" },
        { first: "Jose", last: "Annunziato" },
        ];
        $scope.employees = employees;
        $scope.hello = "Hello From Angular JS";
        $scope.goodbye = "Bye From Angular JS";
        console.log("AngularHelloController");
        });
